<p><strong>Meta Box –∫–ª–∞—Å—ä—Ç</strong> –∏–∑–ø–æ–ª–∑–≤–∞ Singleton pattern:</p>

```php
class QCN_Meta_Box {
    private static $instance = null;
    
    public static function get_instance() {
        if ( null === self::$instance ) {
            self::$instance = new self();
        }
        return self::$instance;
    }
    
    private function __construct() {
        add_action( 'add_meta_boxes', array( $this, 'add_meta_box' ) );
        add_action( 'save_post', array( $this, 'save_note' ), 10, 2 );
    }
    
    public function save_note( $post_id, $post ) {
        // Nonce verification
        if ( ! isset( $_POST['qcn_meta_box_nonce'] ) || 
             ! wp_verify_nonce( $_POST['qcn_meta_box_nonce'], 'qcn_save_note' ) ) {
            return;
        }
        
        // Capability check
        if ( ! current_user_can( 'manage_options', $post_id ) ) {
            return;
        }
        
        // Save note content
        $content = sanitize_textarea_field( wp_unslash( $_POST['qcn_note_content'] ) );
        $color = sanitize_key( $_POST['qcn_note_color'] );
        $status = sanitize_key( $_POST['qcn_note_status'] );
        
        update_post_meta( $post_id, '_qcn_note_content', $content );
        update_post_meta( $post_id, '_qcn_note_color', $color );
        update_post_meta( $post_id, '_qcn_note_status', $status );
        
        // Save to history
        QCN_DB::get_instance()->save_history( 
            $post_id, 
            $content, 
            $color, 
            $status, 
            get_current_user_id() 
        );
        
        // Send notifications if needed
        QCN_Notifications::get_instance()->maybe_send( $post_id, $status );
    }
}

// Initialize
QCN_Meta_Box::get_instance();
```

<p><strong>AJAX endpoint</strong> –∑–∞ quick status change –≤ posts list:</p>

```php
class QCN_Ajax {
    public function __construct() {
        add_action( 'wp_ajax_qcn_quick_status', array( $this, 'quick_status_change' ) );
    }
    
    public function quick_status_change() {
        check_ajax_referer( 'qcn-ajax-nonce', 'nonce' );
        
        if ( ! current_user_can( 'manage_options' ) ) {
            wp_send_json_error( array( 'message' => 'Insufficient permissions' ) );
        }
        
        $post_id = absint( $_POST['post_id'] );
        $status = sanitize_key( $_POST['status'] );
        
        $old_status = get_post_meta( $post_id, '_qcn_note_status', true );
        update_post_meta( $post_id, '_qcn_note_status', $status );
        
        // Notification check
        if ( $status === 'completed' && $old_status !== 'completed' ) {
            do_action( 'qcn_note_completed', $post_id, get_current_user_id() );
        }
        
        wp_send_json_success( array(
            'message' => 'Status updated',
            'status'  => $status
        ) );
    }
}
```
